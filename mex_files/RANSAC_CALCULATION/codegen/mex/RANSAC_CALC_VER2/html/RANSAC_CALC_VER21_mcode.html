  2function [R,T,error,BestFit,State_RANSAC]=RANSAC_CALC_VER2(Ya,Yb,options)
  3%%% Implementation of the RANSAC algorithm of the paper
  4% @INPROCEEDINGS {gpandey-2011b,
  5%     AUTHOR = { Gaurav Pandey and James R. McBride and Silvio Savarese and Ryan M. Eustice },
  6%     TITLE = { Visually Bootstrapped Generalized ICP },
  7%     BOOKTITLE = { Proceedings of the IEEE International Conference on Robotics and Automation },
  8%     YEAR = { 2011 },
  9%     ADDRESS = { Shanghai, China },
 10%     NOTE = { Accepted, To Appear },
 11% }
 12
 13% 1: input: YA, YB, SA, SB,
 14% 2: output: The estimated transformation [R0; t0]
 15% 3: Establish camera constrained SIFT correspondences between
 16% SA and SB.
 17% 4: Store the matches in a list L.
 18% 5: while iter < MAXITER do
 19% 6: Randomly pick 3 pairs of points from the list L.
 20% 7: Retrieve these 3 pair of points from YA and YB.
 21% 8: Calculate the 6-DOF rigid body transformation [R; t]
 22% that best aligns these 3 points.
 23% 9: Store this transformation in an array M, M[iter] =
 24% [R; t]
 25% 10: Apply the transformation to YB to map Scan Bï¿½s
 26% points into the reference frame of Scan A: Y0
 27% B = RYB + t
 28% 11: Calculate the set cardinality of pose-consistent SIFT
 29% correspondences that agree with the current transformation
 30% (i.e., those that satisfy a Euclidean threshold
 31% on spatial proximity): n = j(Y0
 32% B (L) ? YA(L)) < j
 33% 12: Store the number of pose-consistent correspondences
 34% in an array N, N[iter] = n
 35% 13: iter = iter + 1
 36% 14: end while
 37% 15: Find the index i that has maximum number of correspondences
 38% in N.
 39% 16: Retrieve the transformation corresponding to index i
 40% 17: Recalculate the Transformation and Rotation
 41% from M. [R0; T0] = M[i]. This is the required transformation.
 42
 43epsilon = 0.01; %% confidence interval (1-epsilon) is the probability that
 44% we draw a set from the all samples that consists of all inliers
 45nIterations = options.MaxIteration;
 46maxSupport = 5; %% since the size of the set which produces any hypothesis
 47% is 5 for each hypothesis we have at least 5 inliers
 48%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
 49% matches matrix has the following structure
 50% Yb   Currnet Frame    matches First Row
 51% Ya   Previous Frame   matches Second Row
 52%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
 53
 54iter=1;
 55Y0b=zeros(size(Yb));
 56M=struct('R',[],'T',[],'ErrorSum',[]);
 57coder.varsize('M(:)', [1 1200]);
 58Cardinality=zeros(1,options.MaxIteration);
 59ErrorSum = 0;
 60nPoints = size(Yb,2);
 61nSetHypothesisGenerator =5;
 62while iter<min(nIterations,options.MaxIteration) %% this value is updated during each loop running
 63    % 6: Randomly pick 3 pairs of points from the list L.
 64    RandIndex = get_rand(nSetHypothesisGenerator, size(Ya,2) );  %% Remember that indices should not be the same
 65    % 7: Retrieve these 3 pair of points from YA and YB.
 66    SetPoints_a=Ya(:,RandIndex); %% Point sets should be in the form P={P1,P2,..}
 67    SetPoints_b=Yb(:,RandIndex); %%
 68    % 8: Calculate the 6-DOF rigid body transformation [R; t]
 69    % that best aligns these 3 points.
 70    
 71    %% DEBUG comparing the results of current transformation calculation code with the results of quaternion based calculation
 72    [Rot, Trans, State_RANSAC] = find_transform_matrix(SetPoints_a, SetPoints_b);
 73    %     [s, Rot, Trans, err] = absoluteOrientationQuaternion(SetPoints_a, SetPoints_b, 0);
 74    %%
 75    %% DEBUG
 76%     if ~isreal(Rot) || abs(det(Rot)-1)>0.0001
 77%         disp('Rotation matrix contains imaginary values (Degenerate case)')
 78%     end
 79    %    [RR1, TT1] = icp(SetPoints_a',SetPoints_b');
 80    %     [RR2, TT2] = icp(Ya,Yb);
 81    
 82    % 9: Store this transformation in an array M, M[iter] =
 83    % [R; t]
 84    M(iter).R=Rot;
 85    M(iter).T=Trans;
 86    M(iter).ErrorSum=0;
 87    % 10: Apply the transformation to YB to map Scan Bï¿½s
 88    % points into the reference frame of Scan A: Y0B = RYB + t
 89    %%% For later usage we need to keep the support set of each Hypothesis.
 90    %%% This is especially useful in recalculation step
 91    M(iter).SupportSet.Ya=[];
 92    M(iter).SupportSet.Yb=[];
 93    
 94    Y0b=Rot*Yb+repmat(Trans,1,size(Yb,2));
 95    residu = Y0b-Ya;
 96    normResidu = sqrt(residu(1,:).^2+residu(2,:).^2+residu(3,:).^2);
 97    position_inliers = normResidu<options.DistanceThreshold;
 98    Cardinality(iter) = sum(position_inliers);
 99    M(iter).SupportSet.Ya = Ya(:,position_inliers);
100    M(iter).SupportSet.Yb = Yb(:,position_inliers);
101    M(iter).ErrorSum = sum(normResidu(position_inliers));
102    if Cardinality(iter)>=maxSupport
103        maxSupport = Cardinality(iter);
104        nIterations = ceil(log(epsilon)/log( 1-(Cardinality(iter)/nPoints)^nSetHypothesisGenerator ) );
105    end
106    %     for (i=1:size(Ya,2))
107    %         try
108    %         Y0b(:,i)=Rot*Yb(:,i)+Trans;
109    %         catch
110    %             disp('Salam Guguli')
111    %         end
112    %         % 11: Calculate the set cardinality of pose-consistent SIFT
113    %         % correspondences that agree with the current transformation
114    %         % (i.e., those that satisfy an Euclidean threshold
115    %         % on spatial proximity): n = norm(Y0b(L) - YA(L)) < epsilon
116    %         % 12: Store the number of pose-consistent correspondences
117    %         % in an array N, N[iter] = n
118    %         TempError=norm(Y0b(:,i)-Ya(:,i));
119    %         M(iter).ErrorSum = M(iter).ErrorSum+ TempError;
120    %         if TempError<=options.DistanceThreshold
121    %             M(iter).SupportSet.Ya=[M(iter).SupportSet.Ya,Ya(:,i)];
122    %             M(iter).SupportSet.Yb=[M(iter).SupportSet.Yb,Yb(:,i)];
123    %             Cardinality(iter)=Cardinality(iter)+1;
124    %         end
125    %     end
126    iter=iter+1;
127end
128% 15: Find the index "iter" that has maximum number of correspondences
129% in N.
130for iii=1:length(M)
131    eee1(iii)=M(iii).ErrorSum;
132    eee2(iii)=Cardinality(iii);
133end
134MaxCardinality=max(eee2);
135eee2(eee2~=max(eee2))=0;
136eee1(eee2==0)=10000;
137[C,I] = min(eee1);
138BestError=C;
139BestFitIdx=I;
140BestFit=MaxCardinality;
141% [BestFit BestFitIdx]=max(Cardinality);
142% 16: Retrieve the transformation corresponding to index i
143% from M. [R0; T0] = M[i]. This is the required transformation.
144R= M(BestFitIdx).R;
145T= M(BestFitIdx).T;
146% 17:  Recalculate the Transformation and Rotationfor the best hypothesis
147
148
149%% DEBUG
150% This is the original code that I used but it seems  that it fails in some situations (degenerate)
151[R, T, State_RANSAC] = find_transform_matrix(M(BestFitIdx).SupportSet.Ya, M(BestFitIdx).SupportSet.Yb);
152% I want to test using quaternion based calculation
153% [s, R, T, err] = absoluteOrientationQuaternion(M(BestFitIdx).SupportSet.Ya, M(BestFitIdx).SupportSet.Yb, 0);
154%%
155% if ~isreal(R) || abs(det(R)-1)>0.0001
156%     disp('Rotation matrix contains imaginary values (Degenerate case)')
157% end
158
159error=M(BestFitIdx).ErrorSum;
160error.mYa =M(BestFitIdx).SupportSet.Ya;
161error.mYb =M(BestFitIdx).SupportSet.Yb;
162% disp(['max iter = ' num2str(length(M)) ])
163% 18: covariance calculation
164% covariance = covariance_estimate_RANSAC(M(BestFitIdx).SupportSet.Ya, M(BestFitIdx).SupportSet.Yb,R,T);
165
166% covariance = cov_est_RANSAC_deriv(M(BestFitIdx).SupportSet.Ya, M(BestFitIdx).SupportSet.Yb,R,T);
